    export interface RuleMatch {
      ruleId: string;
      file: string;
      line: number;      // 1-based (convert from ast-grep's 0-based)
      column: number;
      text: string;
      message: string;
      severity: string;
    }

    export interface ConventionResult {
      ruleId: string;
      name: string;           // human-readable convention name
      category: string;       // error-handling, imports, async, exports, components, types
      matchingFiles: string[];
      totalApplicableFiles: number;
      adoptionPercent: number; // 0-100
      confidence: "HIGH-CONF" | "MEDIUM-CONF" | "LOW-CONF";
      trend: "Stable";        // always Stable in v1 per D-11
      evidence: Array<{ file: string; line: number; description: string }>;
    }

    export interface CompetingPair {
      a: string;       // ruleId of pattern A
      b: string;       // ruleId of pattern B
      label: string;   // human-readable conflict name
    }

    export interface ConflictInfo {
      label: string;
      patternA: { ruleId: string; name: string; adoption: number };
      patternB: { ruleId: string; name: string; adoption: number };
    }

    export interface GoldenFileEntry {
      filePath: string;
      conventionsFollowed: number;
      conventionsApplicable: number;
      density: number;          // conventionsFollowed / conventionsApplicable
    }

    export interface ConventionScanResult {
      conventions: ConventionResult[];
      conflicts: ConflictInfo[];
      goldenFiles: GoldenFileEntry[];
      totalRulesEvaluated: number;
      totalConventionsDetected: number;
    }

    **Function: `runAstGrepScan(rulesDir: string, targetDir: string): RuleMatch[]`**
    - Verify `sg` is available: `execSync("sg --version", { encoding: "utf-8" })` in try/catch. If not found, throw with message from UI-SPEC: "ast-grep CLI (sg) not found. Convention detection skipped. Install with: npm install -g @ast-grep/cli"
    - Run: `execSync(\`sg scan --rule ${rulesDir} --json ${targetDir}\`, { encoding: "utf-8", maxBuffer: 50 * 1024 * 1024 })`
    - Parse JSON output. ast-grep returns an array of match objects.
    - Map each match to RuleMatch, converting line numbers from 0-based to 1-based (+1 per Pitfall 2)
    - Handle empty output (no matches) gracefully -- return empty array
    - Handle ast-grep returning non-zero exit code -- some rules may not match, that's fine

    **Function: `countApplicableFiles(targetDir: string, language: "TypeScript" | "Python"): number`**
    - Walk targetDir recursively
    - Count files with extensions matching language: `.ts`, `.tsx`, `.js`, `.jsx` for TypeScript; `.py` for Python
    - Skip test files (`*.test.ts`, `*.spec.ts`, `__tests__/`) per convention rule ignores
    - Return count

    **Function: `calculateAdoption(matches: RuleMatch[], totalApplicableFiles: number): { adoptionPercent: number; matchingFiles: string[]; confidence: "HIGH-CONF" | "MEDIUM-CONF" | "LOW-CONF" }`**
    - Group matches by file (unique files matching)
    - adoptionPercent = (unique matching files / totalApplicableFiles) * 100, rounded to nearest integer
    - confidence: if adoptionPercent >= 80 AND totalApplicableFiles >= 10 -> "HIGH-CONF"; if adoptionPercent >= 50 -> "MEDIUM-CONF"; else "LOW-CONF"
    - Return matchingFiles as array of unique file paths

    **Function: `buildEvidence(matches: RuleMatch[], maxEvidence: number = 3): Array<{ file: string; line: number; description: string }>`**
    - Take first `maxEvidence` matches (sorted by file path for determinism)
    - For each: return { file: relative path, line: 1-based line, description: match.message truncated to 80 chars }
