    computeConventionAdherence(options):
    - Input: changedFiles (string[]), conventionsPath (string, path to conventions.md)
    - Reads conventions.md, parses with parseDetectorConventions
    - Filters to HIGH-CONF conventions only
    - For each changed file, counts how many HIGH-CONF conventions the file follows vs total applicable
    - Returns { percent: number (0-100), violatingFiles: number, totalFiles: number }
    - Empty changedFiles returns { percent: 100, violatingFiles: 0, totalFiles: 0 }
    - Missing conventions.md returns { percent: 100, violatingFiles: 0, totalFiles: 0 } (no conventions to violate)

    computeBlastRadiusScore(options):
    - Input: changedFiles (string[]), db (DatabaseType)
    - For each changed file, gets blast radius from graph analytics
    - Normalizes: 0 affected = 100%, 1-5 = 90%, 6-20 = 70%, 21-50 = 50%, 50+ = 30%
    - Averages across changed files
    - Returns { totalAffected: number, normalized: number (0-100), riskBreakdown: Record<string, number> }
    - Empty changedFiles returns { totalAffected: 0, normalized: 100, riskBreakdown: {} }

    computeViolationImpact(options):
    - Input: changedFiles (string[]), violationsPath (string, path to convention-violations.json)
    - Reads violations JSON, counts violations for changed files
    - Normalized: 0 violations = 100%, 1-2 = 85%, 3-5 = 70%, 6-10 = 50%, 10+ = 30%
    - Returns { total: number, byRule: Record<string, number>, normalized: number (0-100) }

    computeImportCorrectness(options):
    - Input: changedFiles (string[]), db (DatabaseType)
    - For each changed file, checks if all import edges resolve to existing nodes
    - Returns { percent: number (0-100), broken: number, total: number }
    - Files not in graph scored as 100% (per Open Question 3)

    computeCompositeScore(conventionAdherence, blastRadius, violationImpact, importCorrectness):
    - All inputs are 0-100 percentages

    - Returns { percent: number, grade: string }

    renderScorecard(scorecard):
    - Takes full DeterministicScorecard object
    - Returns markdown string with table format showing all 6 metrics + composite
    - Includes letter grade prominently at top

    ```typescript
    export function computeCompositeScore(
      conventionAdherence: number,
      blastRadiusNormalized: number,
      violationImpact: number,
      importCorrectness: number,
    ): { percent: number; grade: string } {
      const raw = (conventionAdherence * 0.25) + (blastRadiusNormalized * 0.25) +
                  (violationImpact * 0.25) + (importCorrectness * 0.25);
      const percent = Math.round(Math.max(0, Math.min(100, raw)));
      const grade = percent >= 90 ? "A" : percent >= 85 ? "B+" : percent >= 80 ? "B" :
                    percent >= 70 ? "C+" : percent >= 60 ? "C" : "F";
      return { percent, grade };
    }
    ```
