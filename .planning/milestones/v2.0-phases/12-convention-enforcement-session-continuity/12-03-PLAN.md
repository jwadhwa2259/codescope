    2. Define the shell wrapper script as a template string constant `WRAPPER_SCRIPT`:
    ```sh
    #!/bin/sh
    # CodeScope convention enforcement pre-commit hook
    # Auto-generated by: npx codescope install-hooks
    # Do not edit -- use 'npx codescope uninstall-hooks' to remove

    # Run existing hook if it was preserved
    if [ -f ".git/hooks/pre-commit.codescope-backup" ]; then
      .git/hooks/pre-commit.codescope-backup
      EXISTING_EXIT=$?
      if [ $EXISTING_EXIT -ne 0 ]; then
        exit $EXISTING_EXIT
      fi
    fi

    # Get staged files (Added, Copied, Modified only)
    STAGED_FILES=$(git diff --cached --name-only --diff-filter=ACM)
    if [ -z "$STAGED_FILES" ]; then
      exit 0
    fi

    # Check if sg (ast-grep) is available
    if ! command -v sg >/dev/null 2>&1; then
      if ! npx sg --version >/dev/null 2>&1; then
        echo "[CodeScope] ast-grep (sg) not found -- skipping convention check"
        exit 0
      fi
    fi

    # Run CodeScope convention enforcement
    node "node_modules/codescope/dist/enforcement/pre-commit-check.mjs" $STAGED_FILES
    ```

    3. Define the husky block as constant `HUSKY_BLOCK`:
    ```sh
    # codescope-enforcement-start
    STAGED_FILES=$(git diff --cached --name-only --diff-filter=ACM)
    if [ -n "$STAGED_FILES" ]; then
      node "node_modules/codescope/dist/enforcement/pre-commit-check.mjs" $STAGED_FILES
    fi
    # codescope-enforcement-end
    ```

         - Return `{ installed: true, method: 'husky', backedUp: false, message: 'CodeScope enforcement added to husky pre-commit hook.' }`.
       - **If no husky (D-01 fallback):**
         - Check `.git/hooks/` dir exists. Create with `mkdirSync` if needed.
         - Check if `.git/hooks/pre-commit` exists.
